include_directories(${CMAKE_SOURCE_DIR})

#-------------------------------------------------------------------------------
# EosNsCommon library sources
#-------------------------------------------------------------------------------
set(
  EOS_NS_COMMON_SRCS
  MDException.hh
  Constants.cc   Constants.hh

  # Namespace interface
  interface/IView.hh
  interface/IFileMDSvc.hh
  interface/IContainerMDSvc.hh
  interface/IFileMD.hh
  interface/IContainerMD.hh
  interface/IChLogContainerMDSvc.hh
  interface/IChLogFileMDSvc.hh

  # Namespace utils
  utils/DataHelper.cc
  utils/Descriptor.cc
  utils/ThreadUtils.cc
  utils/Buffer.hh)

add_library(
  EosNsCommon SHARED
  ${EOS_NS_COMMON_SRCS})

target_link_libraries(
  EosNsCommon
  ${Z_LIBRARY})

set_target_properties(
  EosNsCommon
  PROPERTIES
  VERSION ${VERSION}
  SOVERSION ${VERSION_MAJOR}
  INTERFACE_LINK_LIBRARIES "")

install(
  TARGETS EosNsCommon
  LIBRARY DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR}
  ARCHIVE DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR})

#-------------------------------------------------------------------------------
# EosNsCommon-Static library
#-------------------------------------------------------------------------------
if (Linux)
  add_library(
    EosNsCommon-Static STATIC
    ${EOS_NS_COMMON_SRCS})

  target_link_libraries(
    EosNsCommon-Static
    ${Z_LIBRARY_STATIC})

  set_target_properties(
    EosNsCommon-Static
    PROPERTIES
    POSITION_INDEPENDENT_CODE True)
endif ()
